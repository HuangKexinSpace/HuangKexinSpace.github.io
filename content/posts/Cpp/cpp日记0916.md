- 1.5 类
  在头文件person.h中声明类
```
#ifndef PERSON_H // 防止头文件重复包含
#define PERSON_H
#include <string>
class Person {
public:
    // 构造函数，从
    Person(const std::string& name, int age);

    // 获取名字
    std::string getName() const;

    // 获取年龄
    int getAge() const;

    // 设置名字
    void setName(const std::string& name);

    // 设置年龄
    void setAge(int age);

private:
    std::string name;
    int age;
};

#endif // PERSON_H
```
